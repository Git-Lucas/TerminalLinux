./{scriptName}.sh: run the script after grants permissions;
{command} --help: displays possible arguments based on the command;
{scriptName}.sh: run the script if the path is in environment variable;
bash {scriptName}.sh: run the script without the step of grants permissions;
cat {fileName}: displays the content of this file;
chmod {flags} {fileName}: grants permissions;
clear: clears the terminal screen;
cp {fileName} {directoryName}: copy the file to the directory;
echo {text} > {fileName}: override the content of this file with the text informed as parameter; in case of file does not exists in path, create;
echo {text} >> {fileName}: adds the text at the end of the file;
export PATH=$PATH:{pathScript}: add the path in environment variable PATH, and allows execute the script from any folder;
head -c {numberCharacters} {fileName}: displays the firsts numberCharacters of the file;
less {folderName}.zip or .tar.gz: display the content compressed;
ls: list the contents of a directory;
mkdir {folderName} {folderName}? {folderName}?: create a directory;
mv {fileName} {newName or newDirectory}: rename or move the file;
nano: enter text editor;
pwd: displays the current path;
rm {fileName}: remove the file;
rm -r {folderName}: remove all files in the directory and the directory;
rmdir {folderName}: remove the directory;
sorte {fileName}: sort the file (ascending);
tail -n {numberLines} {fileName}: displays the latests lines of the file;
tar -czf {destinationFolderName}.tar.gz {sourceFolder}: create zip file to extension .tar.gz;
tar -xzf {folderName}.tar.gz: extract zip file;
touch {fileName}: create a file;
uniq {fileName}: displays the content of the file, without duplicated lines (in sequence), without update the file;
unzip {folderName}.zip: unzip the folder in current directory with log;
unzip -q {folderName}.zip: unzip the folder in current directory without log;
vi (:/{search}): search the text in the file and position the cursor on the line;
vi (:q!): exit without save;
vi (:q): exit text editor;
vi (:s/{searchText}/{newText}): replace the search text by the new text;
vi (:w {fileName}?): save the file;
vi (:x): save and exit;
vi (A): enable edit mode and position the cursor at the end of the line;
vi (dd): cut the line;
vi (esc): exit edit mode;
vi (i): enable edit mode;
vi (o): insert a line;
vi (p): paste the line;
vi (yy): copy the line;
vi {fileName}?: text editor terminal or open file in text editor terminal;
wc {fileName}: statistics file (number lines, number words, number characters);
zip -r {destinationFolderName}.zip {sourceFolder}: zip source folder;
